条件判断

a = 2
b = 3
if a > b:
	print(a + b)
elif a < b:
	print(b - a)
else:
	print(a , b)
# a == b	a 是否等于 b
# a > b	a 是否大于 b
# a >= b	a 是否大于等于 b
# a < b	a 是否小于 b
# a <= b	a 是否小于等于 b
# a != b	a 是否不等于 b
print(2 < 3 and 2 < 5)                # True
print(2 > 3 or 3 == 3)                # True
print(2 > 3 or not 3 == 3 and 5 < 10) # False
# True and True	需要两边同时满足才能返回 True
# True or False	只要一边是 True 则返回 True
# not True	给出相反结果

循环

# for 循环 天然适合在有限的循环中
# python 截取参数一般是 头实尾虚 包括头不包括尾
for i in range(5):
	print(i) 
# 0 1 2 3 4  range(5) 一个参数, 默认从0,开始
for i in range(2, 5):
	print(i)
# 2 3 4   range(2, 5) 两个参数,头实尾虚
for i in range(3, 10, 2):
  print(i)
# 3 5 7 9 range(3, 10, 2) 三个参数,前面两个参数为范围,最后一个参数,每隔几个数字取一个

# while  可以被用在无限循环中
num = 10
while num != 20:
    num += 1
    print(num)
# 死循环 

#continue 和 break
num = 10
while num != 20:
    num += 1
    if num > 19:
        break
    print(num)
# break 结束当前循环

for i in range(10):
    if i % 2 == 0:
        continue # 跳过偶数
    print(i)
 # continue 跳过此次循环进行下一次循环

数据种类

list 列表

car_list =["A186","宝马","奔驰","劳斯莱斯"]
# 取单个元素
print(car_list[0])   # A186      index 下标首位   
print(car_list[-1])  # 劳斯莱斯  index 尾下标首位  
print(car_list[3])   # 劳斯莱斯  index 0开始到这里3位最后一个元素
print(car_list[-2])  # 奔驰      index 负数从-1尾 从后往前数第二个
# 取某个元素集合
print(car_list[:3])  # ['A186', '宝马', '奔驰']     :号在数字前即为0到3区间,取头不取尾
print(car_list[2:4]) # ['奔驰', '劳斯莱斯']          取头不取尾
print(car_list[-3:]) # ['宝马', '奔驰', '劳斯莱斯']   :号在数字后即为-3到尾,取到最后
# 替换元素值
car_list[1] = 12   
print(car_list[1])
# 列表中，你可以存放不同类型的元素，字符，数字，甚至列表里还能有列表。
car_list[3]=[1,2,4,5]
print(car_list)      # ['A186', '宝马', '奔驰', [1, 2, 4, 5]]

list 常用API

增加元素

  • append() : 在列表末尾添加新元素。
  • extend() : 在列表末尾添加扩展列表中的多个值。
  • insert() : 在指定位置插入一个元素。

删除元素

  • remove() : 指定要删除的元素。
  • pop() : 删除指定位置的元素，如果不指定位置，则默认删除最后一个元素。
  • clear() : 删除列表中的所有元素。

修改元素

  • 直接指定列表中的位置并赋予新的值。

查询元素

  • index() : 查找指定元素的位置。
  • count() : 统计指定元素在列表中出现的次数。

排序

  • sort() : 按升序排序。
  • reverse() : 反转列表。

其他操作

  • len() : 列表的长度。
  • in : 检查元素是否在列表中

Dict 字典

car_dict ={"A186":"白色","宝马":"黑色","奔驰":"蓝色","劳斯莱斯":"暗橘色"}
print(car_dict)
# {'A186': '白色', '宝马': '黑色', '奔驰': '蓝色', '劳斯莱斯': '暗橘色'} 
# 一个key 对应一个value  如果一个Dict中 存在2个相同的key 后面一个key会覆盖前面的key 
# value 可以是任何数据类型 套娃
car_dict["A186"] ="暗紫色"
print(car_dict)
# {'A186': '暗紫色', '宝马': '黑色', '奔驰': '蓝色', '劳斯莱斯': '暗橘色'}
car_dict["A1861"] ="暗黑色"
print(car_dict)
# {'A186': '暗紫色', '宝马': '黑色', '奔驰': '蓝色', '劳斯莱斯': '暗橘色', 'A1861': '暗黑色'}

Dict 常用API

增加元素

  • dict[key] = value : 添加一个键值对。
  • update() : 添加多个键值对。

删除元素

  • del : 删除指定键值对。
  • clear() : 删除字典中的所有元素。

修改元素

  • 直接指定字典中的键并赋予新的值。

查询元素

  • get() : 获取指定键的值。如果键不存在，则返回 None 或提供的默认值。
  • keys() : 返回字典中所有键的列表。
  • values() : 返回字典中所有值的列表。
  • items() : 返回字典中所有键值对的元组。
  • in : 检查键是否在字典中。

其他操作

  • len() : 字典中键值对的个数。

Tuple 元组

cars= ("宝马", "奔驰", ";劳斯莱斯")
print(cars[1])
# 该存储数据,我们就希望它不能改变，特别是你的代码要给别人使用的时候，你没法控制，
# 也不希望他们改变你的这些固定值。
cars[1] ="黑马" # 这里会报错
# TypeError: 'tuple' object does not support item assignment
# 无法变更 无法修改

Tuple 常用API

查询元素

  • count() : 统计指定元素在元组中出现的次数。
  • index() : 查找指定元素的位置。

其他操作

  • len() : 元组的长度。
  • in : 检查元素是否在元组中。

Set 合集

car_dict =({"A186","白色","宝马","黑色","奔驰","蓝色","劳斯莱斯","暗橘色"})
# {'黑色', '宝马', '蓝色', '劳斯莱斯', '暗橘色', 'A186', '白色', '奔驰'}

Set 常用API

增加元素

  • add() : 在集合中添加一个元素，如果元素已存在，则不执行任何操作。
  • update() : 在集合中添加多个元素。

删除元素

  • remove() : 移除指定元素。如果元素不存在，则会引发 KeyError。
  • discard() : 移除指定元素。如果元素不存在，则不执行任何操作。
  • pop() : 移除并返回集合中的任意一个元素。如果集合为空，则引发 KeyError。
  • clear() : 移除集合中的所有元素。

查询元素

  • in : 检查元素是否在集合中。
  • issubset() : 检查集合是否为另一个集合的子集。
  • issuperset() : 检查集合是否为另一个集合的超集。
  • difference() : 返回一个新集合，其中包含只在第一个集合中出现的元素。
  • intersection() : 返回一个新集合，其中包含同时在两个集合中出现的元素。
  • union() : 返回一个新集合，其中包含两个集合中的所有元素。
  • symmetric_difference() : 返回一个新集合，其中包含只出现在其中一个集合中的元素。

其他操作

  • len() : 集合中元素的数量。
  • copy() : 返回集合的副本。

循环中运用

# list 循环遍历
# 1
car_list = ["A186", "宝马", "奔驰", "劳斯莱斯"]
for i in range(len(car_list)):
    print(car_list[i])
# A186  宝马  奔驰  劳斯莱斯
# 2
for i in car_list:
    print(i)	
# A186  宝马  奔驰  劳斯莱斯

# dict 集合循环遍历
car_dict ={"A186":"白色","宝马":"黑色","奔驰":"蓝色","劳斯莱斯":"暗橘色"}
for key in car_dict.keys():
    print("key:", key)
# key: A186  key: 宝马  key: 奔驰  key: 劳斯莱斯
for value in car_dict.values():
    print("value:", value)
# value: 白色  value: 黑色  value: 蓝色  value: 暗橘色
for value,key in car_dict.items():
    print("key:", key, ", value:", value)
# key: 白色 , value: A186  key: 黑色 , value: 宝马  key: 蓝色 , value: 奔驰
# key: 暗橘色 , value: 劳斯莱斯
