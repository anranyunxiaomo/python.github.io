条件判断

a = 2
b = 3
if a > b:
	print(a + b)
elif a < b:
	print(b - a)
else:
	print(a , b)
# a == b	a 是否等于 b
# a > b	a 是否大于 b
# a >= b	a 是否大于等于 b
# a < b	a 是否小于 b
# a <= b	a 是否小于等于 b
# a != b	a 是否不等于 b
print(2 < 3 and 2 < 5)                # True
print(2 > 3 or 3 == 3)                # True
print(2 > 3 or not 3 == 3 and 5 < 10) # False
# True and True	需要两边同时满足才能返回 True
# True or False	只要一边是 True 则返回 True
# not True	给出相反结果

循环

# for 循环 天然适合在有限的循环中
# python 截取参数一般是 头实尾虚 包括头不包括尾
for i in range(5):
	print(i) 
# 0 1 2 3 4  range(5) 一个参数, 默认从0,开始
for i in range(2, 5):
	print(i)
# 2 3 4   range(2, 5) 两个参数,头实尾虚
for i in range(3, 10, 2):
  print(i)
# 3 5 7 9 range(3, 10, 2) 三个参数,前面两个参数为范围,最后一个参数,每隔几个数字取一个

# while  可以被用在无限循环中
num = 10
while num != 20:
    num += 1
    print(num)
# 死循环 

#continue 和 break
num = 10
while num != 20:
    num += 1
    if num > 19:
        break
    print(num)
# break 结束当前循环

for i in range(10):
    if i % 2 == 0:
        continue # 跳过偶数
    print(i)
 # continue 跳过此次循环进行下一次循环

数据种类

list 列表

carList =["A186","宝马","奔驰","劳斯莱斯"]
# 取单个元素
print(carList[0])   # A186      index 下标首位   
print(carList[-1])  # 劳斯莱斯  index 尾下标首位  
print(carList[3])   # 劳斯莱斯  index 0开始到这里3位最后一个元素
print(carList[-2])  # 奔驰      index 负数从-1尾 从后往前数第二个
# 取某个元素集合
print(carList[:3])  # ['A186', '宝马', '奔驰']     :号在数字前即为0到3区间,取头不取尾
print(carList[2:4]) # ['奔驰', '劳斯莱斯']          取头不取尾
print(carList[-3:]) # ['宝马', '奔驰', '劳斯莱斯']   :号在数字后即为-3到尾,取到最后
# 替换元素值
carList[1] = 12   
print(carList[1])
# 列表中，你可以存放不同类型的元素，字符，数字，甚至列表里还能有列表。
carList[3]=[1,2,4,5]
print(carList)      # ['A186', '宝马', '奔驰', [1, 2, 4, 5]]

list常用API

### 增加元素

- `append()` : 在列表末尾添加新元素。
- `extend()` : 在列表末尾添加扩展列表中的多个值。
- `insert()` : 在指定位置插入一个元素。

### 删除元素

- `remove()` : 指定要删除的元素。
- `pop()` : 删除指定位置的元素，如果不指定位置，则默认删除最后一个元素。
- `clear()` : 删除列表中的所有元素。

### 修改元素

- 直接指定列表中的位置并赋予新的值。

### 查询元素

- `index()` : 查找指定元素的位置。
- `count()` : 统计指定元素在列表中出现的次数。

### 排序

- `sort()` : 按升序排序。
- `reverse()` : 反转列表。

### 其他操作

- `len()` : 列表的长度。
- `in` : 检查元素是否在列表中
